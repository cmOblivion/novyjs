/**
 * 一个uuid创建器
 * @func uuid
 * @returns {Function} 一个获取uuid的函数
 * @example
 * let uuidMaker = uuid();
 * let concreteUuid = uuidMaker();
 */
function uuid(){
  let tid = 0;
  return () => {
    return tid++;
  }
}

export { uuid };
